## API Reference

To use this component in your project, include the following import statement in your file.

```tsx
import {
  AlertDialog,
  AlertDialogBackdrop,
  AlertDialogContent,
  AlertDialogHeader,
  AlertDialogCloseButton,
  AlertDialogFooter,
  AlertDialogBody,
} from "@/components/ui/alert-dialog"

```

```tsx
export default () => (
  export default () => (
  <AlertDialog>
    <AlertDialogBackdrop />
    <AlertDialogContent>
      <AlertDialogHeader>
        <AlertDialogCloseButton />
      </AlertDialogHeader>
      <AlertDialogBody />
      <AlertDialogFooter />
    </AlertDialogContent>
  </AlertDialog>
);
);
```
