## API Reference

To use this component in your project, include the following import statement in your file.

```jsx

import {
  Calendar,
  CalendarHeader,
  CalendarHeaderNext,
  CalendarHeaderPrev,
  CalendarHeaderTitle,
  CalendarGrid,
  CalendarGridWeek,
  CalendarGridDays,
} from '@/components/ui/calendar';

```
```jsx
export default () => (
  <Calendar
    value={new Date()}
    onChange={(date) => console.log(date)}
  >
    <CalendarHeader>
      <CalendarHeaderPrev />
      <CalendarHeaderTitle />
      <CalendarHeaderNext />
    </CalendarHeader>
    <CalendarGrid>
      <CalendarGridWeek />
      <CalendarGridDays />
    </CalendarGrid>
  </Calendar>
);
```
